## This animation shows the simulation of the death of two species with
##   certain probabilities.
library(animation)
library(SNPRelate)
library(gdsfmt)
library(BiocInstaller)
library(reshape2)
library(MASS)
library(ggplot2)
library(gapminder)
library(plyr)
ani.options(interval = 0.3, nmax = ifelse(interactive(), 
    25, 2))
par(ann = FALSE, mar = rep(0, 4))
ecol.death.sim(nr = 10, nc = 10, num.sp = c(50, 50), col.sp = c(8, 
    2), pch.sp = c(20, 17), col.die = 1, pch.die = 4, cex = 3)
